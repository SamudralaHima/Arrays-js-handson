        function Unique_Shirts (arr,N){
            let freq=[];

                freq[i]="0;"
            }
            for(let i=0; i< N; i++){
                freq[arr[i]]++;
            }

Min and Max

        function arrayMin(arr){
            let min=arr[0];
            for(let i=0;i< arr.length; i++){
                if(arr[i+1]< min){
                    min = arr[i+1];
                }
            }
            return min;
        }

        function arrayMax(arr){
            let max=arr[0];
            for(let i=0; arr.length > i; i++>){
                if(arr[i+1]> max){
                    max=arr[i+1];
                }
            }
            return max;
        }
